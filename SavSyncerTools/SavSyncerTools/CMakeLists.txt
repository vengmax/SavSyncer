cmake_minimum_required(VERSION 3.16)
project(SavSyncerTools LANGUAGES CXX)

include(qt.cmake)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(QT NAMES Qt6 Qt5 REQUIRED COMPONENTS Core)
find_package(Qt${QT_VERSION_MAJOR}
    COMPONENTS
        Core
)
qt_standard_project_setup()

set(PROJECT_SOURCES
    main.cpp
)

qt_add_executable(SavSyncerTools ${PROJECT_SOURCES})

set_target_properties(SavSyncerTools
    PROPERTIES
        WIN32_EXECUTABLE FALSE
)

target_link_libraries(SavSyncerTools
    PUBLIC
        Qt::Core
)
